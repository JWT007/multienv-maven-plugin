22  /**
23   * This mojo will get the main artifact of the current project unpack it and use the files of the appropriate
24   * environment and produce new files which contain the original files and the supplemental files which have been given
25   * by the configuration for each environment.
26   * 
27   * @author Karl-Heinz Marbaise <a href="mailto:khmarbaise@soebes.de">khmarbaise@soebes.de</a>
28   */
29  @Mojo( name = "environment", defaultPhase = LifecyclePhase.PACKAGE, requiresProject = true, threadSafe = true )
30  public class EnvironmentMojo
31      extends AbstractMultiEnvMojo
32  {
33  
34      /**
35       * The JAR archiver needed for archiving the environments.
36       */
37      @Component( role = Archiver.class, hint = "jar" )
38      private JarArchiver jarArchiver;
39  
40      @Component
41      private ArchiverManager manager;
42  
43      public void execute()
44          throws MojoExecutionException, MojoFailureException
45      {
46          String[] identifiedEnvironments = getTheEnvironments( getSourceDirectory() );
47  
48          if ( identifiedEnvironments.length == 0 )
49          {
50              getLog().warn( "No Environment folders found." );
51              return;
52          }
53  
54          createLoggingOutput( identifiedEnvironments );
55  
56          String archiveExt = getArchiveExtensionOfTheProjectMainArtifact();
57  
58          File unpackFolder = createUnpackFolder();
59  
60          File resourceResult = createPluginResourceOutput();
61  
62          filterResources( resourceResult );
63  
64          // Currently we use the main artifact of the project
65          // TODO: May be should make this configurable? So we might use any kind of artifact
66          // as source?
67          unarchiveFile( getMavenProject().getArtifact().getFile(), unpackFolder, archiveExt );
68  
69          for ( String environment : identifiedEnvironments )
70          {
71              getLog().info( "Building Environment: '" + environment + "'" );
72  
73              // Check why this can happen?
74              if ( environment.isEmpty() )
75              {
76                  getLog().warn( "The given folder '" + environment + "' is empty." );
77                  continue;
78              }
79  
80              try
81              {
82                  File targetFolder = new File( resourceResult, environment );
83                  File createArchiveFile = createArchiveFile( unpackFolder, targetFolder, environment, archiveExt );
84                  getProjectHelper().attachArtifact( getMavenProject(), getMavenProject().getPackaging(), environment,
85                                                     createArchiveFile );
86              }
87              catch ( NoSuchArchiverException e )
88              {
89                  getLog().error( "Archive creation failed.", e );
90              }
91              catch ( IOException e )
92              {
93                  getLog().error( "IO Exception.", e );
94              }
95          }
96  
97      }
98  
99      private void createGZIPArchive( String includes )
100         throws IOException, NoSuchArchiverException
101     {
102         try
103         {
104             File baseFolder = new File( getSourceDirectory(), includes );
105             File theOriginalFile = new File( baseFolder, "first.properties" );
106             Archiver gzipArchiver = manager.getArchiver( "gzip" );
107 
108             gzipArchiver.addFile( theOriginalFile, "first.properties.gz" );
109 
110             gzipArchiver.setDestFile( new File( baseFolder, "first.properties.gz" ) );
111             gzipArchiver.createArchive();
112         }
113         catch ( ArchiverException e )
114         {
115             getLog().error( "Archive creation failed.", e );
116         }
117 
118     }
119 
120     private void unarchiveFile( File sourceFile, File destDirectory, String archiveExt )
121         throws MojoExecutionException
122     {
123         try
124         {
125             UnArchiver unArchiver = manager.getUnArchiver( archiveExt );
126 
127             unArchiver.setSourceFile( sourceFile );
128             unArchiver.setUseJvmChmod( true );
129             unArchiver.setDestDirectory( destDirectory );
130             unArchiver.setOverwrite( true );
131             unArchiver.extract();
132         }
133         catch ( ArchiverException e )
134         {
135             throw new MojoExecutionException( "Error unpacking file [" + sourceFile.getAbsolutePath() + "]" + " to ["
136                 + destDirectory.getAbsolutePath() + "]", e );
137         }
138         catch ( NoSuchArchiverException e )
139         {
140             getLog().error( "Unknown archiver." + " with unknown extension [" + archiveExt + "]" );
141         }
142     }
143 
144     private File createArchiveFile( File unpackFolder, File targetFolder, String folder, String archiveExt )
145         throws NoSuchArchiverException, IOException, MojoExecutionException
146     {
147         final MavenArchiver mavenArchiver = new MavenArchiver();
148 
149         mavenArchiver.setArchiver( jarArchiver );
150 
151         jarArchiver.addFileSet( new DefaultFileSet( targetFolder ) );
152         jarArchiver.addFileSet( new DefaultFileSet( unpackFolder ) );
153         // jarArchiver.setDuplicateBehavior( duplicate );
154 
155         File resultArchive = getArchiveFile( getOutputDirectory(), getFinalName(), folder, archiveExt );
156 
157         mavenArchiver.setOutputFile( resultArchive );
158         try
159         {
160             mavenArchiver.createArchive( getMavenSession(), getMavenProject(), getArchive() );
161         }
162         catch ( ArchiverException | ManifestException | DependencyResolutionRequiredException e )
163         {
164             getLog().error( e.getMessage(), e );
165             throw new MojoExecutionException( e.getMessage(), e );
166         }
167 
168         return resultArchive;
169 
170     }
